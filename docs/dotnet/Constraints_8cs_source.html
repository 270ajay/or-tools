358
360 public CumulativeConstraint AddDemand<D>(IntervalVar interval, D demand)
361 {
362 CumulativeConstraintProto cumul = Proto.Cumulative;
363 cumul.Intervals.Add(interval.GetIndex());
364 LinearExpr demandExpr = cp_model_.GetLinearExpr(demand);
365 cumul.Demands.Add(cp_model_.GetLinearExpressionProto(demandExpr));
366 return this;
367 }
368
370 public CumulativeConstraint AddDemands<D>(IEnumerable<IntervalVar> intervals, IEnumerable<D> demands)
371 {
372 CumulativeConstraintProto cumul = Proto.Cumulative;
373 foreach (var p in intervals.Zip(demands, (i, d) => new { Interval = i, Demand = d })) {
374 cumul.Intervals.Add(p.Interval.GetIndex());
375 LinearExpr demandExpr = cp_model_.GetLinearExpr(p.Demand);
376 cumul.Demands.Add(cp_model_.GetLinearExpressionProto(demandExpr));
377 }
378 return this;
379 }

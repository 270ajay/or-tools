156  public long Value(LinearExpr e)
157  {
158  List<Term> terms = new List<Term>();
159  terms.Add(new Term(e, 1));
160  long constant = 0;
161 
162  while (terms.Count > 0)
163  {
164  Term term = terms[0];
165  terms.RemoveAt(0);
166  if (term.coefficient == 0)
167  {
168  continue;
169  }
170 
171  if (term.expr is LinearExprBuilder a)
172  {
173  constant += term.coefficient * a.Offset;
174  foreach (Term sub in a.Terms)
175  {
176  if (term.coefficient == 1)
177  {
178  terms.Add(sub);
179  }
180  else
181  {
182  terms.Add(new Term(sub.expr, sub.coefficient * term.coefficient));
183  }
184  }
185  }
186  else if (term.expr is IntVar intVar)
187  {
188  int index = intVar.GetIndex();
189  long value = index >= 0 ? response_.Solution[index] : -response_.Solution[-index - 1];
190  constant += term.coefficient * value;
191  }
192  else if (term.expr is NotBoolVar)
193  {
194  throw new ArgumentException("Cannot evaluate a literal in an integer expression.");
195  }
196  else
197  {
198  throw new ArgumentException("Cannot evaluate '" + term.expr.ToString() + "' in an integer expression");
199  }
200  }
201  return constant;
202  }

27 bool VbpParser::ParseFile(const std::string& data_filename) {
28  vbp_.Clear();
29 
30  load_status_ = DIMENSION_SECTION;
31  for (const std::string& line : FileLines(data_filename)) {
32  if (load_status_ == ERROR_FOUND) break;
33  ProcessLine(line);
34  }
35 
36  // Checks status.
37  if (load_status_ == ERROR_FOUND) {
38  LOG(INFO) << vbp_.DebugString();
39  return false;
40  }
41  return vbp_.item_size() == num_declared_items_;
42 }

49 void VbpParser::ProcessLine(const std::string& line) {
50  const std::vector<std::string> words =
51  absl::StrSplit(line, absl::ByAnyChar(" :\t\r"), absl::SkipEmpty());
52 
53  if (words.empty()) return;
54 
55  switch (load_status_) {
56  case NOT_STARTED: {
57  LOG(FATAL) << "Should not be here";
58  }
59  case DIMENSION_SECTION: {
60  if (words.size() != 1) {
61  ReportError(line);
62  return;
63  }
64  num_resources_ = strtoint32(words[0]);
65  load_status_ = BIN_SECTION;
66  break;
67  }
68  case BIN_SECTION: {
69  if (words.size() != num_resources_) {
70  ReportError(line);
71  return;
72  }
73  for (const std::string& dim_str : words) {
74  vbp_.add_resource_capacity(strtoint64(dim_str));
75  }
76  load_status_ = NUMBER_OF_ITEMS_SECTION;
77  break;
78  }
79  case NUMBER_OF_ITEMS_SECTION: {
80  if (words.size() != 1) {
81  ReportError(line);
82  return;
83  }
84  num_declared_items_ = strtoint32(words[0]);
85  load_status_ = ITEM_SECTION;
86  break;
87  }
88  case ITEM_SECTION: {
89  if (words.size() != num_resources_ + 1) {
90  ReportError(line);
91  return;
92  }
93  Item* const item = vbp_.add_item();
94  for (int i = 0; i < num_resources_; ++i) {
95  item->add_resource_usage(strtoint64(words[i]));
96  }
97  item->set_num_copies(strtoint32(words[num_resources_]));
98  item->set_max_number_of_copies_per_bin(item->num_copies());
99  break;
100  }
101  case ERROR_FOUND: {
102  break;
103  }
104  }
105 }
